Table Utilizador as U {
  id int [pk, increment]
  nome varchar
  password varchar
}

Table Notificacao {
  id int [pk]
  nome varchar 
  idUtilizador int [ref: > Utilizador.id]
}

Table Prova {
  id int [pk, increment]
  idEpoca int [ref: > Epoca.ano]
  nomeProva varchar
  nomeCircuito varchar
  data datetime
  horaProva time
  idNotificacao int [ref: > Notificacao.id]
  idLocalizacao varchar [ref: > Localizacao.id]
  }

Table Epoca {
  ano int [pk]
}

Table Localizacao {
  id varchar [pk]
  nome varchar
  nacionalidade varchar [ref: > Pais.nome]
  latitude float
  longitude float
}

Table Qualificacao {
  idProva int [pk, ref: > Prova.id]
  idPiloto varchar [ref: > Piloto.id]
  posicaoQualificacao int
  qualificao1 time
  qualificao2 time
  qualificao3 time
}

Table Resultado {
  idProva int [pk, ref: > Prova.id]
  idPiloto varchar [ref: > Piloto.id]
  posicaoFinal int
  posicaoInicial int
  tempo time
  pontos int
  estado varchar
}

Table Piloto {
  id varchar [pk]
  nome varchar
  numero int
  dataNascimento date
  nacionalidade varchar [ref: > Pais.nome]
  equipaId int [ref: > Equipa.id]
}

Table Equipa {
  id int [pk,increment]
  nome varchar
  nacionalidade varchar [ref: > Pais.nome]
  idEpoca int [ref: > Epoca.ano]
  
}

Table Pais {
  nome varchar [pk]
  nacionalidade varchar
}
